#include <iostream> 
 
int count(int r, int p[][2])
{
    int count=0, i=0, j=0;
 
    for(i=0; i<5; i++){
                if(r*r>(p[i][0]*p[i][0]+p[i][1]*p[i][1])){
                count++;
                std::cout<<"\nCounted for the point number "<<i<<". Count is: "<<count;
                std::cout<<"\nPoint inside the circle is: ("<<p[i][0]<<","<<p[i][1]<<")";
                }
    }
    std::cout<<"\nThe total number of points inside circle is: "<<count<<std::endl;
    return count;
}
