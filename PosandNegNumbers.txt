void posAndNeg(vector<int> &arr){
    vector<int>pos;
    vector<int>neg;
    for(auto x: arr){
        if(x>=0){
            pos.push_back(x);
        }
        else{
            neg.push_back(x);
        }
    }

    int n=arr.size()/2;
    arr.clear();
    for(int i=0; i<n; i++){
        arr.push_back(pos[i]);
        arr.push_back(neg[i]);
    }
}
