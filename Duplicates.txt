#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int x = 12344677;
    int n;
    cout<<"The size of x is: "<<sizeof(x)<<endl;
    n = x/10;
    cout<<"\n x/10 is: "<<n<<endl;
    n = x%10;
    cout<<"\n x%10 is: "<<n<<endl;
    int j=10, size=0;
    vector <int> a;
    while(j>0){
        j = x%10;
        cout<<"\n x%10 is: "<<j;
        a.push_back(j);
        x=x/10;
        size++;
    }
    int n1 = sizeof(a)/sizeof(int);
    cout<<"\nsize of n1 is: "<<n1;
    cout<<"\nThe size is: "<<size;
    for(int i=0; i<size-1; i++){
        cout<<"\n Vector element: "<<a[i];
    }
    
    for(int i=0; i<size-1; i++){
        for(j=i+1; j<size-1; j++){
            if(a[i]==a[j]){
                cout<<"\nDuplicate element is: "<<a[i];
            }
        }
    }
    
    return 0;
}
